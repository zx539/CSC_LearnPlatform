### 三、实操案例：二维矢量场的旋度与散度计算
#### （一）案例场景
选取通信工程电磁学中常见的**二维平面矢量场**（简化的电场/磁场模型），计算其旋度和散度，直观观察矢量场的物理特性，理解旋度、散度在电磁场分析中的实际应用。

#### （二）代码实现（Python）
代码适配你“结合数理基础理解专业内容”的认知风格，每一步均标注数学原理与电磁学关联，可直接运行验证。

```python
import numpy as np
import matplotlib.pyplot as plt

# ---------- 1. 定义二维矢量场（贴合电磁学场景，如简化平面电场） ----------
# 矢量场F = (Fx, Fy)，Fx为x方向分量，Fy为y方向分量，对应电磁场的分量形式
def vector_field(x, y):
    # 示例：模拟平面静电场的简化形式，Fx与x成正比，Fy与y成正比（可替换为其他电磁场模型，如涡旋场）
    Fx = 2 * x  # x方向分量，对应电场的径向分量简化
    Fy = 3 * y  # y方向分量，对应电场的切向分量简化
    return Fx, Fy

# ---------- 2. 计算散度（散度公式：div F = ∂Fx/∂x + ∂Fy/∂y，对应电磁场的有源/无源特性） ----------
def calculate_divergence(x, y):
    # 利用偏导数计算散度，与你已掌握的高等数学偏导数知识直接衔接
    # 对Fx关于x求偏导，对Fy关于y求偏导，求和得到散度
    dFx_dx = 2  # ∂Fx/∂x = ∂(2x)/∂x = 2
    dFy_dy = 3  # ∂Fy/∂y = ∂(3y)/∂y = 3
    divergence = dFx_dx + dFy_dy  # 散度公式直接应用
    return divergence

# ---------- 3. 计算旋度（二维旋度公式：rot F = ∂Fy/∂x - ∂Fx/∂y，对应电磁场的涡旋特性） ----------
def calculate_curl(x, y):
    # 二维旋度仅存在z方向分量，公式为∂Fy/∂x - ∂Fx/∂y，是电磁学中判断场是否为涡旋场的核心依据
    dFy_dx = 0  # ∂Fy/∂x = ∂(3y)/∂x = 0（Fy与x无关，偏导数为0）
    dFx_dy = 0  # ∂Fx/∂y = ∂(2x)/∂y = 0（Fx与y无关，偏导数为0）
    curl_z = dFy_dx - dFx_dy  # 二维旋度z分量计算
    return curl_z

# ---------- 4. 可视化与结果分析（直观理解物理意义，贴合电磁学应用场景） ----------
def visualize_field(x_range, y_range, num_points=20):
    # 生成网格点数据，对应电磁场分析的空间采样
    x = np.linspace(x_range[0], x_range[1], num_points)
    y = np.linspace(y_range[0], y_range[1], num_points)
    X, Y = np.meshgrid(x, y)
    
    # 计算矢量场分量
    Fx, Fy = vector_field(X, Y)
    
    # 计算各点的散度和旋度
    divergence_values = np.zeros_like(X)
    curl_values = np.zeros_like(X)
    for i in range(num_points):
        for j in range(num_points):
            divergence_values[i, j] = calculate_divergence(x[j], y[i])
            curl_values[i, j] = calculate_curl(x[j], y[i])
    
    # 绘制矢量场与散度热力图（散度反映场的发散程度，对应电磁场的源/汇）
    plt.figure(figsize=(12, 5))
    
    # 子图1：矢量场分布
    plt.subplot(1, 2, 1)
    plt.quiver(X, Y, Fx, Fy, color='blue', alpha=0.7)
    plt.title('二维矢量场分布（简化电磁场模型）')
    plt.xlabel('x方向')
    plt.ylabel('y方向')
    plt.grid(True)
    
    # 子图2：散度热力图
    plt.subplot(1, 2, 2)
    plt.contourf(X, Y, divergence_values, levels=20, cmap='coolwarm')
    plt.colorbar(label='散度值（对应电磁场发散强度）')
    plt.title('矢量场散度分布')
    plt.xlabel('x方向')
    plt.ylabel('y方向')
    plt.grid(True)
    
    plt.tight_layout()
    plt.show()
    
    # 输出关键结论（贴合电磁学知识应用）
    print("散度计算结论：")
    print(f"本案例矢量场的散度为常数：{calculate_divergence(0, 0)}，说明该场为“有源场”（对应电磁学中电荷产生的电场，散度与电荷密度正相关）")
    print(f"旋度计算结论：")
    print(f"本案例矢量场的旋度为常数：{calculate_curl(0, 0)}，说明该场为“无旋场”（对应静电场的保守性，无涡旋特性）")

# ---------- 5. 运行主函数 ----------
if __name__ == "__main__":
    # 定义观测范围，对应电磁场分析的空间区域
    x_range = (-5, 5)
    y_range = (-5, 5)
    visualize_field(x_range, y_range)
```
